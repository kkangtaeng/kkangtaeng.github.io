      });
      setFilter(filtered);
    }
  } else if (free) {
    setfreeName(free.target.textContent);
    if (free.target.textContent === "모든 체험") {
      setFilter(services);
    } else {
      let filtered = services.filter((service) => {
        if (free.target.textContent === "유") {
          return service.demo === true;
        } else {
          return service.demo === false;
        }
      });
      setFilter(filtered);
    }
  }
};
