### 힘든 점

- 필터기능을 구현하는게 조금 힘들었다.. 3가지 기준에서 각 항목을 전부 적용해야 했기에 접근방법이 어려웠던것 같다. 처음에는 각 항목을 순차적으로 필터링하여 각 기준이 겹쳐지게 되어 나중에 모든가격같은 모두보기 항목을 클릭했을 때에는 전체서비스에서 필터링이 되는것이 아닌 필터링이 이미 된 배열을 필터링하고 있었다..
- 그러다가 문득 그럼 그 기준에서의 항목을 상태로 관리하면 어떨까? 그럼 그 상태를 기준으로 계속 필터링을 하면 되지않을까? 라는 생각이 들었고 바로 수정작업을 했다!!

```jsx
const filterServices = (category, price, free) => {
  if (category) {
    setCategoryName(category.target.textContent);
    if (category.target.textContent === "모든 카테고리") {
      setSelect({ ...select, category: "" });
    } else {
      setSelect({
        ...select,
        category: category.target.textContent.toLowerCase(),
      });
    }
  } else if (price) {
    setpriceName(price.target.textContent);
    if (price.target.textContent === "모든 가격") {
      setSelect({ ...select, price: "" });
    } else {
      setSelect({
        ...select,
        price: Number(price.target.textContent),
      });
    }
  } else if (free) {
    setfreeName(free.target.textContent);
    if (free.target.textContent === "모든 체험") {
      setSelect({ ...select, free: "" });
    } else {
      let boolean;
      if (free.target.textContent === "유") {
        boolean = true;
      } else {
        boolean = false;
      }
      setSelect({
        ...select,
        free: boolean,
      });
    }
  }
};
```

- 그래서 처음 작성했던 코드에서 위처럼 상태로 관리하였고, 그 후 바로 위 함수안에서 filter를 하니 select 상태가 undefined로 나왔다.. 생각해보니 코드가 먼저 실행되고 나중에 적용되니 undefined라는것을 바로 깨달았다ㅠㅠ

```jsx
useEffect(() => {

      } else if (select.free === "") {

          service.category === select.category &&
```
