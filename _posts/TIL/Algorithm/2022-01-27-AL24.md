### 풀이

```jsx
function solution(people, limit) {
  let sorted = people.sort((a, b) => a - b);
  let large = sorted.length - 1;
  let small = 0;
  let count = 0;
  while (large > small) {
    if (sorted[large] + sorted[small] <= limit) {
      count++;
      small++;
      large--;
    } else {
      large--;
    }
  }
  return people.length - count;
}
```
