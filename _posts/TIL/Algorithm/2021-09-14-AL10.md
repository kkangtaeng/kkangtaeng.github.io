### 풀이

전부 소문자로 변경 후 각 개수를 구하고 같은지 다른지 확인한다!

```jsx
function solution(s){
    let lowerCase = s.toLowerCase()
    let smallp = 0
    let smally = 0
    for(let n = 0 ; n < s.length ; n++) {
        if(lowerCase[n] === 'p') {
            smallp++
        } else if(lowerCase[n] === 'y') {
            smally++
        }
    }
    if(smally === smallp) {
        return true
    } else {
        return false
    }
}
```
