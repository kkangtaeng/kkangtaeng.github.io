### 풀이

```jsx
function solution(nums) {
  let result = [];
  for (let i = 0; i < nums.length - 2; i++) {
    for (let j = i + 1; j < nums.length - 1; j++) {
      for (let k = j + 1; k < nums.length; k++) {
        let div = nums[i] + nums[j] + nums[k];
        let prime = () => {
          for (let n = 2; n < div; n++) {
            if (div % n === 0) {
              return false;
            }
          }
          return true;
        };
        if (prime(div)) {
          result.push(div);
        }
      }
    }
  }
  return result.length;
}
```
