### 풀이

```jsx
function solution(a, b) {
  let week = ["FRI", "SAT", "SUN", "MON", "TUE", "WED"];
  let days = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
  let month = 0;
  for (let i = 0; i < a - 1; i++) {
    month += days[i];
  }
  month += b;
  let div = month % 7;
  if (div === 0) {
    return "THU";
  } else {
    return week[div - 1];
  }
}
```

- 해당 일에 7 나머지가 0 이면 목요일이다..
