### 풀이

```jsx
function solution(numbers) {
  let result = [];
  let sorted = numbers.sort((a, b) => a - b);
  for (let n = 0; n < sorted.length - 1; n++) {
    for (let m = n + 1; m < sorted.length; m++) {
      let plus = sorted[n] + sorted[m];
      if (result.includes(plus)) {
        continue;
      } else {
        result.push(plus);
      }
    }
  }
  return result.sort((a, b) => a - b);
}
```
