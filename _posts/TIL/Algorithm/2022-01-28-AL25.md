### 풀이

```jsx
function solution(array, commands) {
  let result = [];
  for (let i = 0; i < commands.length; i++) {
    let arr = [];
    for (let j = commands[i][0] - 1; j < commands[i][1]; j++) {
      arr.push(array[j]);
    }
    // let sorted = arr.sort((a, b) => a - b);
    result.push(arr.sort((a, b) => a - b)[commands[i][2] - 1]);
  }
  return result;
}
```
