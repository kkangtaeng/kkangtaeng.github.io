### 풀이

```jsx
function solution(arr1, arr2) {
  let result = []
  for(let i = 0 ; i < arr1.length ; i++) {
      let arr = []
      for(let n = 0 ; n < arr2[0].length ; n++) {
        let sum = 0
          for(let m = 0 ; m < arr1[0].length ; m++) {
              sum += arr1[i][m] * arr2[m][n]
          }
          arr.push(sum)
      }
      result.push(arr)
  }
  return result
}
```
