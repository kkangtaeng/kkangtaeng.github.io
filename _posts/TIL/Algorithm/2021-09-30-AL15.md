### 풀이

```jsx
function solution(A,B){
    let sortA = A.sort((a,b) => a - b)
    let sortB = B.sort((a,b) => b - a)
    let result = 0
    for(let n = 0 ; n < A.length ; n++) {
        result += sortA[n] * sortB[n]
    }
    return result
}
```
