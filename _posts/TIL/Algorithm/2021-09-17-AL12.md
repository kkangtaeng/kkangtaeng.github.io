### 풀이

```jsx
function solution(s) {
    let result = ''
    result += s[0].toUpperCase()
    for(let n = 1 ; n < s.length ; n++) {
        if(s[n-1] === ' ') {
            result += s[n].toUpperCase()
        }
        else {
            result += s[n].toLowerCase()
        }
    }
    return result
}
```
